-- with model in mvc

public class UserViewModel
{
    public List<Getalluser> Listgetalluser { get; set; }
}

[HttpGet]
public JsonResult GetAllUsers()
{
    Getalluser _getalluser = new Getalluser();

    UserViewModel model = new UserViewModel();
    model.Listgetalluser = _getalluser.Getallusers();

    return Json(model, JsonRequestBehavior.AllowGet);
}
